import java.io.*;
import java.util.*;

public class Main{

public static void main(String[] args) throws Exception {
    // write your code here
    Scanner scn=new Scanner(System.in);
    int n=scn.nextInt();
    int old_zero=1;
    int old_one=1;
    for(int i=2;i<=n;i++)
    {
        int new_one=old_zero+old_one;
        int new_zero=old_one;
        old_zero=new_zero;
        old_one=new_one;
    }
    System.out.println(old_zero+old_one);
 }

}
